\begin{frame}[fragile]{How do I install ITensor/PastaQ?}

\begin{enumerate}[<+->]

  \item Download Julia: \myhref{https://julialang.org/downloads/}{julialang.org/downloads}
  \item Launch Julia:
\begin{lstlisting}[language=JuliaLocal, numbers=none, style=julia, basicstyle=\scriptsize\ttfamily]
$ julia
\end{lstlisting}
  \item Type the following commands:
\begin{lstlisting}[language=JuliaLocal, numbers=none, style=julia, basicstyle=\scriptsize\ttfamily]
julia> using Pkg

julia> Pkg.add("ITensors")
[...]

julia> Pkg.add("PastaQ")
[...]
\end{lstlisting}

\end{enumerate}

\end{frame}
