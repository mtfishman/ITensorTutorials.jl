function op(
  ::OpName"U",
  ::SiteType"S=1/2";
  $\theta$1, $\theta$2
)
  c1 = cos($\theta$1/2)
  s1 = sin($\theta$1/2)
  c2 = cos($\theta$2/2)
  s2 = sin($\theta$2/2)
  return [
    c1  0   0 -s1
    0  c2 -s2   0
    0  s2  c2   0
    s1  0   0  c1
  ]
end
\end{lstlisting}
