$\psi_0$ = Zp1 * Zp2

function E($\theta$)
  $\theta$1, $\theta$2 = $\theta$
  U$_\theta$ = op("U", i1, i2;
              $\theta$1=$\theta$1, $\theta$2=$\theta$2)
  $\psi_\theta$ = apply(U$_\theta$, $\psi_0$)
  return inner($\psi_\theta$', H, $\psi_\theta$)
end
\end{lstlisting}

\end{onlyenv}
