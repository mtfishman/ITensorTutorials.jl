\begin{onlyenv}<1->
\begin{lstlisting}[language=JuliaLocal, style=julia, basicstyle=\small]
H = ITensor(i1', i2', i1, i2)
H[i1'=>2, i2'=>1,
   i1=>2, i2=>1] = -1
# …
\end{lstlisting}
\end{onlyenv}

\begin{onlyenv}<2->
\begin{lstlisting}[language=JuliaLocal, style=julia, basicstyle=\small]
Id1 = op("Id", i1)
Z1 = op("Z", i1)
X1 = op("X", i1)
# …

ZZ = Z1 * Z2
XI = X1 * Id2
IX = Id1 * X2

h = 0.5
H = -ZZ + h * (XI + IX)
\end{lstlisting}
\end{onlyenv}

\end{column}
