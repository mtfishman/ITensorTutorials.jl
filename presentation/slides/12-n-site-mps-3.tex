\begin{lstlisting}[language=JuliaLocal, style=julia, mathescape, basicstyle=\small]
function ising(n; h)
  H = OpSum()
  for j in 1:(n - 1)
    H -= "Z", j, "Z", j + 1
  end
  for j in 1:n
    H += h, "X", j
  end
  return H
end
\end{lstlisting}

\end{onlyenv}
