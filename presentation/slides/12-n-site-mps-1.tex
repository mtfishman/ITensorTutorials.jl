  i = [Index(2, "S=1/2")
        for j in 1:n]

  Zp = MPS(i, "Z+")
  Zm = MPS(i, "Z-")

  Cat = (Zp + Zm)/√2
 \end{lstlisting}

  norm(Cat)

  inner(Zp, Cat)
\end{lstlisting}
